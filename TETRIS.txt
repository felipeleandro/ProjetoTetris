CREATE DATABASE TETRIS;

CREATE TABLE PONTUACAO
(
    NOME VARCHAR (50),
    PONTUACAO INT,
    NIVEL INT,
    TEMPO_PARTIDA TIME,
    LINHAS_ELIMINADAS INT
);

CREATE TABLE usuarios
(
nome VARCHAR (40),
data_nascimento date,
telefone varchar (30),
cpf varchar (30),
email varchar (40),
senha varchar (50)
);
